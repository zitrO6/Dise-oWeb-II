Referencia Rápida de Borders

/* Border completo (width style color) */
border: 2px solid #6c5ce7;

/* Bordes individuales */
border-top: 1px solid #6c5ce7;
border-right: 2px dashed #00b894;
border-bottom: 3px dotted #fd79a8;
border-left: 4px double #636e72;

/* Border-radius */
border-radius: 10px; /* Todos los lados */
border-radius: 10px 20px 30px 40px; /* TL TR BR BL */
border-radius: 50%; /* Círculo */

/* Solo algunas esquinas */
border-top-left-radius: 10px;
border-top-right-radius: 10px;
border-bottom-right-radius: 10px;
border-bottom-left-radius: 10px;

/* Estilos: solid | dashed | dotted | double | groove | ridge | inset | outset | none */
